#include<iostream>
using namespace std;
int main(){
    int n,i,j;
    cin>>n;
    for(i=1;i<=n;i++){
        cout<<endl;
        for(j=1;j<=n;j++){
            if(j==1 || i==1 || i==n || j==n){
                cout<<"*";
            }else{
                cout<<" ";
            }
            }
            
    }
    return 0;
}
